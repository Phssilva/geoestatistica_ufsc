Análise da Precipitação Anual em Pelotas - RS¶

Este notebook realiza análises probabilísticas da precipitação anual em Pelotas - RS, considerando que os dados seguem uma distribuição normal com média de 1280 mm e desvio padrão de 290 mm.

# Parâmetros da distribuição normal
media = 1280  # mm
desvio_padrao = 290  # mm

# Criando um objeto de distribuição normal
distribuicao = stats.norm(loc=media, scale=desvio_padrao)

# Definindo o intervalo para plotagem
x = np.linspace(media - 4*desvio_padrao, media + 4*desvio_padrao, 1000)
y = distribuicao.pdf(x)

# Plotando a distribuição
plt.figure(figsize=(10, 6))
plt.plot(x, y, 'b-', linewidth=2)
plt.axvline(media, color='r', linestyle='--', alpha=0.7, label=f'Média: {media} mm')
plt.axvline(media + desvio_padrao, color='g', linestyle='--', alpha=0.7, label=f'Média + DP: {media + desvio_padrao} mm')
plt.axvline(media - desvio_padrao, color='g', linestyle='--', alpha=0.7, label=f'Média - DP: {media - desvio_padrao} mm')
plt.title('Distribuição Normal da Precipitação Anual em Pelotas - RS')
plt.xlabel('Precipitação Anual (mm)')
plt.ylabel('Densidade de Probabilidade')
plt.grid(True, alpha=0.3)
plt.legend()
plt.show()

# Percentis para análise
percentis = [5, 20, 40, 95]
valores_precipitacao = {}

# Calculando os valores de precipitação para cada percentil
for p in percentis:
    valor = distribuicao.ppf(p/100)
    valores_precipitacao[p] = valor
    print(f"Precipitação esperada em {p}% do tempo: {valor:.2f} mm")

# Visualização
plt.figure(figsize=(12, 8))
plt.plot(x, y, 'b-', linewidth=2)

cores = ['red', 'orange', 'green', 'purple']
for i, p in enumerate(percentis):
    valor = valores_precipitacao[p]
    plt.axvline(valor, color=cores[i], linestyle='--', alpha=0.7, label=f'{p}% ({valor:.2f} mm)')
    
plt.title('Precipitação Total Anual Esperada em Diferentes Percentis')
plt.xlabel('Precipitação Anual (mm)')
plt.ylabel('Densidade de Probabilidade')
plt.grid(True, alpha=0.3)
plt.legend()
plt.show()

# Visualização
plt.figure(figsize=(10, 6))
plt.plot(x, y, 'b-', linewidth=2)
plt.fill_between(x[(x >= valor_inferior_20) & (x <= valor_superior_20)], 
                 y[(x >= valor_inferior_20) & (x <= valor_superior_20)], 
                 color='orange', alpha=0.5)
plt.axvline(valor_inferior_20, color='r', linestyle='--', alpha=0.7, 
            label=f'Percentil {percentil_inferior_20}%: {valor_inferior_20:.2f} mm')

            label=f'Percentil {percentil_superior_20}%: {valor_superior_20:.2f} mm')

             xy=((valor_inferior_20 + valor_superior_20)/2, distribuicao.pdf((valor_inferior_20 + valor_superior_20)/2)),
